Generates site containing all the partners.

def partners_list(request):

    

  

  

  

    

      

        #
      

      
    

    

      
    partners = Partner.objects.all().order_by('name')

    context = {
        'partners': partners,
    }

    return render(request, 'partners_index.html', context)
